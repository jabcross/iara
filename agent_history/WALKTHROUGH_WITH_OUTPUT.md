```bash
from tools.experiment_framework import load_experiments_yaml, get_parameter_combinations

combos = get_parameter_combinations(config, "regression")

print(f"Total combinations: {len(combos)}\n")

# Group by scheduler
by_scheduler = {}
for combo in combos:
    sched = combo['scheduler']
    if sched not in by_scheduler:
        by_scheduler[sched] = []
    by_scheduler[sched].append(combo)

print("Breakdown by scheduler:")
for sched in sorted(by_scheduler.keys()):
    count = len(by_scheduler[sched])
    print(f"  {sched:15} -> {count:2} instances")

print(f"\nFirst 3 combinations:")
for i, combo in enumerate(combos[:3], 1):
    print(f"  {i}. {combo}")
EOF
```

```bash
$ python3 << 'EOF'
import re

# Extract instance names
names = [
    "05-cholesky_sequential_matrix_size_2048_num_blocks_1",
    "05-cholesky_omp-for_matrix_size_2048_num_blocks_2",
    "08-sift_vf-omp_image_size_512",
    "13-degridder_sequential_size_1024_radius_2"
]

print("Instance Name Breakdown:")
print("=" * 80)

for name in names:
    # Split on underscores
    parts = name.split('_')

    # Find where scheduler ends (heuristic)
    print(f"\n{name}")
    print(f"  Pattern: <app>_<scheduler>_<param1>_<value1>_<param2>_<value2>...")

    # Reconstruct
    app = parts[0]  # First part
    sched = parts[1]  # Second part
    params = ' '.join([f"{parts[i]}={parts[i+1]}" for i in range(2, len(parts), 2)])

    print(f"  App:        {app}")
    print(f"  Scheduler:  {sched}")
    print(f"  Parameters: {params}")
EOF
```
